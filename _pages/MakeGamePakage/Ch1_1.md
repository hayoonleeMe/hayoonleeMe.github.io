```cpp
namespace jm
{
    class RotatingStarExample : public Game2D
    {
        float time = 0.0f;

    public:
        void update() override
        {
            rotate(time * 180.0f);
            drawFilledStar(Colors::gold, 0.4f, 0.25f);	// 별을 그린다

            time += this->getTimeStep();
        }

    };
}
```
